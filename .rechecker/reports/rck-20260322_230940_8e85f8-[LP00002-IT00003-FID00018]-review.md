### BUG: `evictStaleBuildResults` does not await cleanup promises
**Severity**: high
**Location**: `evictStaleBuildResults`
**Problem**: The `evictStaleBuildResults` function collects cleanup promises in `cleanupPromises` but then calls `Promise.all(cleanupPromises).catch(...)` without `await`. This means the function returns before the cleanup operations (file system removals) are guaranteed to complete. If the process exits shortly after `evictStaleBuildResults` is called, some temporary build directories might not be cleaned up, leading to disk space accumulation. While the `setInterval` is `unref()`'d, this is still a race condition for any cleanup triggered by `buildPlugin` or if the process is shutting down.
**Fix**: The `Promise.all` call should be `await`ed to ensure all cleanup operations complete before the function finishes.

```typescript
// /Users/emanuelesabetta/ai-maestro/.claude/worktrees/rck-8e85f8/services/plugin-builder-service.ts
function evictStaleBuildResults(): void {
  const now = Date.now()
  // Collect all cleanup promises so they are tracked (not fire-and-forget)
  const cleanupPromises: Promise<void>[] = []

  for (const [id, result] of buildResults) {
    const age = now - new Date(result.createdAt).getTime()
    if (age > BUILD_TTL_MS) {
      buildResults.delete(id)
      const buildDir = path.join(BUILDS_DIR, id)
      cleanupPromises.push(fs.rm(buildDir, { recursive: true, force: true }).catch(() => {}))
    }
  }

  // If still over limit, evict oldest
  if (buildResults.size > MAX_BUILD_RESULTS) {
    const entries = [...buildResults.entries()]
      .sort((a, b) => new Date(a[1].createdAt).getTime() - new Date(b[1].createdAt).getTime())
    const toRemove = entries.slice(0, entries.length - MAX_BUILD_RESULTS)
    for (const [id] of toRemove) {
      buildResults.delete(id)
      const buildDir = path.join(BUILDS_DIR, id)
      cleanupPromises.push(fs.rm(buildDir, { recursive: true, force: true }).catch(() => {}))
    }
  }

  // Execute all cleanup promises concurrently; errors are already handled per-promise above
  // Await the Promise.all to ensure cleanup completes before the function returns.
  Promise.all(cleanupPromises).catch(err => {
    console.error('Error during build result eviction cleanup:', err)
  })
}
```
